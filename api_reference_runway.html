Runway API Reference

Python examples and parameter reference for video generation. Based on Runway API docs. Install: pip install runwayml

1. Image-to-Video (basic)

from runwayml import RunwayML

client = RunwayML(api_key="YOUR_RUNWAY_API_KEY")

task = client.image_to_video.create(
    model="gen4_turbo",
    prompt_image="https://example.com/your-image.jpg",
    prompt_text="A cute bunny hopping in a meadow",
    duration=10,
)

result = task.wait_for_task_output(timeout=600)
video_url = result.output[0]
print(f"Video ready: {video_url}")

2. Image-to-Video (full parameters)

from runwayml import RunwayML

client = RunwayML(api_key="YOUR_RUNWAY_API_KEY")

# Image: HTTPS URL, Runway URI (runway://...), or data URI (data:image/...)
image_uri = "data:image/png;base64,iVBORw0KGgo..."  # or URL

task = client.image_to_video.create(
    model="gen4_turbo",
    prompt_image=image_uri,
    prompt_text="Camera slowly pans across the scene, soft natural light.",
    ratio="720:1280",       # Output resolution
    duration=8,             # 2–10 seconds
    seed=42,                # Optional: for reproducibility
    # content_moderation={"public_figure_threshold": "low"},  # Optional
)

result = task.wait_for_task_output(timeout=600)
video_url = result.output[0]

3. Image-to-Video with prompt images array

# Use array format when specifying position (e.g. first frame)
task = client.image_to_video.create(
    model="gen4_turbo",
    prompt_image=[
        {"uri": "https://example.com/frame.jpg", "position": "first"}
    ],
    prompt_text="Subtle movement, cinematic.",
    ratio="1280:720",
    duration=5,
)

4. Text-to-Video (no image input)

# Text-to-video: model options differ (no gen3a_turbo, gen4_turbo)
task = client.text_to_video.create(
    model="veo3.1",
    prompt_text="A cute bunny hopping in a meadow at sunset",
    ratio="1280:720",
    duration=8,
)

result = task.wait_for_task_output(timeout=600)
video_url = result.output[0]

5. Poll task status manually

task = client.image_to_video.create(
    model="gen4_turbo",
    prompt_image=image_uri,
    prompt_text="Gentle wind in the trees",
    duration=5,
)

# Poll until done
import time
while task.status not in ("SUCCEEDED", "FAILED"):
    time.sleep(5)
    task = client.tasks.retrieve(task.id)

if task.status == "SUCCEEDED":
    video_url = task.output[0]
